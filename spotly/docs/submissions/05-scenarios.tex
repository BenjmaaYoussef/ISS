% ─── Introduction ────────────────────────────────────────────────────────────
This document presents the main use case scenarios of the Spotly platform.
Each scenario is assigned an identifier (S1, S2, \ldots) and is broken down
into numbered steps (S\textit{n}.1, S\textit{n}.2, \ldots). For each step
the involved \textbf{user role} and the relevant \textbf{feature ID(s)} from
the Features \& Roles document are indicated.

Scenarios S1--S8 cover the core Phase~1 features. Scenarios S9--S11 cover
the AI-powered Phase~2 features.

\subsection*{S1 --- Guest Explores the Platform Without an Account}
\begin{mdframed}[style=scenariobox]
\textbf{Actor:} Guest \quad
\textbf{Goal:} Discover available venues and browse the digital menu
without registering.
\end{mdframed}

\begin{longtable}{>{\bfseries}l L{6.5cm} l L{3cm}}
    \toprule

    \midrule
    S1.1 & Guest opens the application and lands on the landing page. & Guest & F2 \\

           without logging in. & Guest & F2 \\
    S1.3 & Guest browses the venue list and applies an activity filter. & Guest & F2 \\

\subsection*{S2 --- New Client Registers and Completes a Booking}
\begin{mdframed}[style=scenariobox]
\textbf{Actor:} Client \quad
\textbf{Goal:} Create an account and submit a table reservation at a
chosen venue.
\end{mdframed}

\begin{longtable}{>{\bfseries}l L{6.5cm} l L{3cm}}
    \toprule
    Step & Description & Role & Features \\
    \midrule
    S2.1 & Guest clicks \emph{Get Started} on the landing page and is
           redirected to the authentication page. & Guest & F1 \\
    S2.2 & Guest fills in first name, last name, email, and password,
           then clicks \emph{Register}. A session is created and the
           user is redirected to the home page as a Client. & Guest $\to$ Client & F1 \\
    S2.3 & Client searches for a venue by name using the search bar. & Client & F2 \\
    S2.4 & Client opens the venue profile page. & Client & F3 \\
    S2.5 & Client clicks \emph{Book a Table} to start the booking flow. & Client & F5 \\
    S2.6 & Client views available environments and selects one. & Client & F5 \\
    S2.7 & Client views the floor plan, picks an available table,
           and selects a date and time. & Client & F6 \\
    S2.8 & Client fills in guest details (name, email, phone, number of
           guests, optional notes) and clicks \emph{Submit Reservation}. & Client & F7 \\
    S2.9 & Client is redirected to the awaiting screen showing the
           animated status timeline. & Client & F8 \\
    \bottomrule
\end{longtable}

\subsection*{S3 --- Admin Processes a Reservation Request}
\begin{mdframed}[style=scenariobox]
\textbf{Actor:} Admin and Client \quad
\textbf{Goal:} Admin approves a pending reservation; client is notified
in real time.
\end{mdframed}

\begin{longtable}{>{\bfseries}l L{6.5cm} l L{3cm}}
    \toprule
    Step & Description & Role & Features \\
    \midrule
    S3.1 & Admin logs in with their credentials and is redirected to
           the admin dashboard. & Admin & F1, F12 \\
    S3.2 & Admin notices a pending reservation count on the dashboard
           and navigates to the reservation queue. & Admin & F12, F15 \\
    S3.3 & Admin reads the reservation details: guest name, date, time,
           environment, table, and number of guests. & Admin & F15 \\
    S3.4 & Admin clicks \emph{Approve}; the reservation status changes
           to \texttt{APPROVED} and a log entry is created. & Admin & F15 \\
    S3.5 & The client's awaiting screen detects the status change via
           the cross-tab storage event and the timeline advances to
           \emph{Approved} without a page refresh. & Client & F8 \\
    S3.6 & Client is automatically redirected to the client dashboard. & Client & F8, F9 \\
    \bottomrule
\end{longtable}

\subsection*{S5 --- Admin Onboards a New Venue}
\begin{mdframed}[style=scenariobox]

    \toprule

\subsection*{S7 --- Staff Handles an Evening Shift}
\begin{mdframed}[style=scenariobox]
\textbf{Actor:} Staff \quad
\textbf{Goal:} Run floor operations: check in booked guests, handle a
walk-in, and acknowledge a waiter call.
\end{mdframed}

\begin{longtable}{>{\bfseries}l L{6.5cm} l L{3cm}}
    \toprule
    Step & Description & Role & Features \\
    \midrule
    S7.1 & Staff logs in and opens the staff operations dashboard. & Staff & F1, F19 \\
    S7.2 & Staff selects the current date and the active environment
           to load the live floor map. & Staff & F19 \\
    S7.3 & A guest with a confirmed reservation arrives; staff clicks
           their table on the floor map and selects \emph{Check In}.
           Status changes to \texttt{CHECKED\_IN}. & Staff & F19 \\
    S7.4 & A second table's reservation guest does not arrive; staff
           marks the table as \emph{No-Show}. Status changes to
           \texttt{NO\_SHOW}. & Staff & F19 \\
    S7.5 & A walk-in party of two arrives; staff clicks \emph{Walk-In
           Guest}, fills in basic details, and a reservation is created
           instantly with status \texttt{CHECKED\_IN}. & Staff & F19 \\
    S7.6 & A table indicator pulses gold on the floor map indicating a
           waiter call; staff clicks \emph{Acknowledge} to dismiss it. & Staff & F20 \\
    S7.7 & At end of service, staff marks all remaining occupied tables
           as \emph{Checked Out}. Statuses change to \texttt{COMPLETED}. & Staff & F19 \\
    \bottomrule
\end{longtable}

\subsection*{S8 --- Admin Rejects a Reservation}
\begin{mdframed}[style=scenariobox]
\textbf{Actor:} Admin and Client \quad
\textbf{Goal:} Admin rejects an unsuitable request; client is notified
and prompted to rebook.
\end{mdframed}

\begin{longtable}{>{\bfseries}l L{6.5cm} l L{3cm}}
    \toprule
    Step & Description & Role & Features \\
    \midrule
    S8.1 & Admin opens the reservation queue and locates a pending
           request that cannot be accommodated. & Admin & F15 \\
    S8.2 & Admin clicks \emph{Reject}; a confirmation dialog appears. & Admin & F15 \\
    S8.3 & Admin confirms; the reservation status changes to
           \texttt{REJECTED} and a log entry is created. & Admin & F15 \\
    S8.4 & The client's awaiting screen detects the rejection via the
           storage event and displays a graceful decline card. & Client & F8 \\
    S8.5 & Client clicks \emph{Try Again}; the booking flow restarts
           from environment selection with the same venue pre-selected. & Client & F8, F5 \\
    \bottomrule
\end{longtable}

\subsection*{S9 --- Client Books via Voice Command}
\begin{mdframed}[style=scenariobox]
\textbf{Actor:} Client \quad
\textbf{Goal:} Use the AI voice booking feature to pre-fill a reservation
form by speaking naturally.

\begin{longtable}{>{\bfseries}l L{6.5cm} l L{3cm}}
    \toprule
    Step & Description & Role & Features \\
    \midrule
    S9.1 & Logged-in client sees the \emph{Voice Booking} section on
           the home page and taps the microphone button. & Client & F21 \\
    S9.2 & The button animates to indicate listening; client speaks:
           \emph{``Book a table for two at Noir for Friday evening.''} & Client & F21 \\
    S9.3 & The system sends the transcript to the AI along with venue

    S9.4 & AI returns extracted intent: venue, date, time, and guest
           count. The confirmation form is pre-filled. & Client & F21 \\
    S9.5 & Client reviews the pre-filled form and clicks \emph{Submit
           Reservation}. & Client & F7 \\
    S9.6 & Client is redirected to the awaiting screen. & Client & F8 \\

\subsection*{S10 --- Client Discovers Venues via Natural Language Description}
\begin{mdframed}[style=scenariobox]
\textbf{Actor:} Client \quad
\textbf{Goal:} Use the AI semantic discovery mode to find a venue that
matches a described atmosphere.
\end{mdframed}

\begin{longtable}{>{\bfseries}l L{6.5cm} l L{3cm}}
    \toprule
    Step & Description & Role & Features \\
    \midrule
    S10.1 & Client opens the home page and switches the discovery toggle
            from \emph{Search} to \emph{Describe}. & Client & F22 \\

    S10.6 & Client starts the booking flow from the venue profile. & Client & F5 \\

\subsection*{S11 --- Admin Generates AI Insights About Reservation Patterns}
\begin{mdframed}[style=scenariobox]
\textbf{Actor:} Admin \quad
\textbf{Goal:} Use the AI insights card to understand booking trends and
act on them.
\end{mdframed}

\begin{longtable}{>{\bfseries}l L{6.5cm} l L{3cm}}
    \toprule
    Step & Description & Role & Features \\
    \midrule
    S11.1 & Admin opens the admin dashboard. & Admin & F12 \\
    S11.2 & Admin scrolls to the \emph{AI Insights} card and clicks
            \emph{Generate}. Skeleton rows appear while the AI processes. & Admin & F23 \\
    S11.3 & The system aggregates the last 30~days of reservation data
            (totals, no-show rate, peak hours, day-of-week distribution)
            and sends it to the AI. & Admin & F23 \\
    S11.4 & AI returns 3--5 actionable observations, each with an icon
            and severity level. The card displays them as a list. & Admin & F23 \\
    S11.5 & Admin reads an insight flagging a high Friday no-show rate
            and decides to send reminder messages before future Friday
            reservations. & Admin & F23 \\
    \bottomrule
\end{longtable}
